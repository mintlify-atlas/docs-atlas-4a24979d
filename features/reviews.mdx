## Overview

The review system allows users to rate and review healthcare facilities based on their experiences. Users can only review facilities where they've had completed appointments, ensuring authentic feedback.

## Viewing Reviews

### Reviews Component

Reviews are displayed on facility detail pages:

```jsx Resenias.jsx
export default function Resenias({ resenias, promedioEstrellas, totalResenias, loading }) {
  if (loading) {
    return <div className="loading-resenias">Cargando reseñas...</div>;
  }

  if (!resenias || resenias.length === 0) {
    return (
      <div className="no-resenias">
        <p>📝 Aún no hay reseñas para este establecimiento</p>
        <p className="no-resenias-subtitle">Sé el primero en dejar tu opinión</p>
      </div>
    );
  }

  return (
    <div className="resenias-container">
      <div className="resenias-header">
        <h3>Reseñas ({totalResenias})</h3>
        <div className="promedio-container">
          <div className="promedio-numero">{promedioEstrellas}</div>
          <Estrellas puntuacion={Math.round(promedioEstrellas)} />
          <div className="promedio-texto">
            Promedio de {totalResenias} {totalResenias === 1 ? 'reseña' : 'reseñas'}
          </div>
        </div>
      </div>
      
      <div className="resenias-lista">
        {resenias.map((resenia) => (
          <ReseniaItem key={resenia.id} resenia={resenia} />
        ))}
      </div>
    </div>
  );
}
```

### Star Display Component

The `Estrellas` component renders star ratings:

```jsx
const Estrellas = ({ puntuacion, size = 'medium' }) => {
  const stars = [];
  for (let i = 1; i <= 5; i++) {
    stars.push(
      <span
        key={i}
        className={`star ${i <= puntuacion ? 'filled' : 'empty'} ${size}`}
      >
        ★
      </span>
    );
  }
  return <div className="estrellas">{stars}</div>;
};
```

## Interactive Star Selector

### SelectorEstrellas Component

The star selector provides hover effects and click handling:

```jsx
const SelectorEstrellas = ({ puntuacion, onChange }) => {
  const [hover, setHover] = useState(0);

  return (
    <div className="selector-estrellas">
      {[1, 2, 3, 4, 5].map((star) => (
        <button
          key={star}
          type="button"
          className={`star-button ${star <= (hover || puntuacion) ? 'filled' : 'empty'}`}
          onClick={() => onChange(star)}
          onMouseEnter={() => setHover(star)}
          onMouseLeave={() => setHover(0)}
          aria-label={`${star} ${star === 1 ? 'estrella' : 'estrellas'}`}
        >
          ★
        </button>
      ))}
      <span className="puntuacion-texto">
        {puntuacion > 0 ? `${puntuacion} de 5 estrellas` : 'Selecciona una puntuación'}
      </span>
    </div>
  );
};
```

## Form Validation

### Client-Side Validation

The form validates all required fields before submission:

```jsx
const handleSubmit = async (e) => {
  e.preventDefault();
  setError(null);

  // Validations
  if (!turnoSeleccionado) {
    setError('Debe seleccionar un turno');
    return;
  }
  if (puntuacion === 0) {
    setError('Debe seleccionar una puntuación');
    return;
  }
  if (!comentario.trim()) {
    setError('Debe escribir un comentario');
    return;
  }
  if (comentario.trim().length < 10) {
    setError('El comentario debe tener al menos 10 caracteres');
    return;
  }

  // Submit review
  await reseniasService.crearResenia(
    turnoSeleccionado,
    establecimientoId,
    puntuacion,
    comentario.trim()
  );
};
```

## Loading Reviews

### useResenias Hook

The custom hook fetches reviews for a specific facility:

```jsx useResenias.js
export const useResenias = (establecimientoId) => {
  const [resenias, setResenias] = useState([]);
  const [loading, setLoading] = useState(true);
  const [promedioEstrellas, setPromedioEstrellas] = useState(0);
  const [totalResenias, setTotalResenias] = useState(0);

  useEffect(() => {
    if (!establecimientoId) return;

    const fetchResenias = async () => {
      try {
        const data = await reseniasService.getReseniasByEstablecimiento(establecimientoId);
        setResenias(data.resenias);
        setPromedioEstrellas(data.promedio);
        setTotalResenias(data.total);
      } catch (error) {
        console.error('Error fetching reviews:', error);
      } finally {
        setLoading(false);
      }
    };

    fetchResenias();
  }, [establecimientoId]);

  return { resenias, loading, promedioEstrellas, totalResenias };
};
```

## Appointment-Review Connection

### useTurnosParaReseniar Hook

This hook fetches only appointments eligible for review:

```jsx
export const useTurnosParaReseniar = (establecimientoId) => {
  const [turnos, setTurnos] = useState([]);
  const [loading, setLoading] = useState(true);
  const { user } = useAuth();

  useEffect(() => {
    if (!user?.mail || !establecimientoId) return;

    const fetchTurnos = async () => {
      try {
        const allTurnos = await fetchMisTurnos(user.mail);
        
        // Filter: completed appointments at this facility without reviews
        const turnosElegibles = allTurnos.filter(turno => {
          const turnoFecha = new Date(turno.datetime || turno.fecha);
          const esDelEstablecimiento = turno.establecimientoId === establecimientoId;
          const esPasado = turnoFecha < new Date();
          const noTieneResenia = !turno.resenia;
          
          return esDelEstablecimiento && esPasado && noTieneResenia;
        });
        
        setTurnos(turnosElegibles);
      } finally {
        setLoading(false);
      }
    };

    fetchTurnos();
  }, [user, establecimientoId]);

  return { turnos, loading };
};
```

<Note>
Users can only review each appointment once. Appointments with existing reviews are filtered out.
</Note>
