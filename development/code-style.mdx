## Overview

This guide outlines the coding standards and conventions used in SaludMap. Following these guidelines ensures consistency and maintainability across the codebase.

---

## General Principles

### Code Quality

1. **Readability First**: Code is read more than it's written
2. **Keep It Simple**: Avoid unnecessary complexity
3. **DRY (Don't Repeat Yourself)**: Extract common logic
4. **YAGNI (You Aren't Gonna Need It)**: Don't add features speculatively
5. **SOLID Principles**: Follow object-oriented design principles

### Naming Conventions

- **Be Descriptive**: Names should clearly indicate purpose
- **Avoid Abbreviations**: Use full words unless commonly understood
- **Consistent Terminology**: Use the same terms throughout codebase


### TypeScript Best Practices

#### Type Annotations

```typescript
// Always use type annotations for function parameters and return types
function calculateTotal(price: number, quantity: number): number {
  return price * quantity;
}

// Use interfaces for object shapes
interface User {
  id: number;
  name: string;
  email: string;
}

function getUser(id: number): Promise<User> {
  // Implementation
}

// Avoid 'any' - use specific types
// Bad
function processData(data: any): any {
  return data;
}

// Good
function processData(data: UserData): ProcessedResult {
  return { processed: true, data };
}
```

#### Null Handling

```typescript
// Use optional chaining and nullish coalescing
const userName = user?.profile?.name ?? 'Guest';
const userAge = user?.age ?? 0;

// Explicit null checks for clarity
if (user === null || user === undefined) {
  throw new Error('User not found');
}

// Better - use optional chaining
if (!user) {
  throw new Error('User not found');
}
```

### Async/Await

```typescript
// Prefer async/await over promises
// Good
async function fetchUserData(id: number): Promise<User> {
  try {
    const user = await prisma.user.findUnique({ where: { id } });
    if (!user) {
      throw new Error('User not found');
    }
    return user;
  } catch (error) {
    console.error('Error fetching user:', error);
    throw error;
  }
}

// Avoid callback hell
// Bad
function fetchUserData(id, callback) {
  database.query('SELECT * FROM users', (err, result) => {
    if (err) callback(err);
    else {
      processUser(result, (err, processed) => {
        if (err) callback(err);
        else callback(null, processed);
      });
    }
  });
}
```

### Error Handling

```typescript
// Use try-catch for async operations
async function createAppointment(data: CreateAppointmentDto) {
  try {
    return await prisma.turno.create({ data });
  } catch (error) {
    if (error.code === 'P2002') {
      throw new ConflictException('Appointment already exists');
    }
    throw new InternalServerErrorException('Failed to create appointment');
  }
}

// Create custom error classes
class ValidationError extends Error {
  constructor(message: string, public field: string) {
    super(message);
    this.name = 'ValidationError';
  }
}

// Use error boundaries in React
class ErrorBoundary extends React.Component {
  componentDidCatch(error, errorInfo) {
    console.error('Error caught:', error, errorInfo);
  }
}
```

### Controllers

```typescript
// Keep controllers thin - delegate to services
@Controller('users')
export class UsersController {
  constructor(private readonly usersService: UsersService) {}

  @Get()
  @UseGuards(JwtAuthGuard)
  async findAll(): Promise<User[]> {
    return await this.usersService.findAll();
  }

  @Post()
  @UsePipes(new ValidationPipe())
  async create(@Body() createUserDto: CreateUserDto): Promise<User> {
    return await this.usersService.create(createUserDto);
  }

  // Use descriptive HTTP methods and status codes
  @Delete(':id')
  @HttpCode(HttpStatus.NO_CONTENT)
  async remove(@Param('id') id: number): Promise<void> {
    await this.usersService.remove(id);
  }
}
```

### Services

```typescript
// Services contain business logic
@Injectable()
export class UsersService {
  constructor(private prisma: PrismaClient) {}

  // Document complex logic
  /**
   * Creates a new user with hashed password
   * @param userData User registration data
   * @returns Created user without password
   * @throws ConflictException if email already exists
   */
  async create(userData: CreateUserDto): Promise<User> {
    // Check for existing user
    const existing = await this.prisma.usuario.findUnique({
      where: { mail: userData.mail },
    });

    if (existing) {
      throw new ConflictException('Email already registered');
    }

    // Hash password
    const hashedPassword = await bcrypt.hash(userData.contrasenia, 10);

    // Create user
    const user = await this.prisma.usuario.create({
      data: {
        ...userData,
        contrasenia: hashedPassword,
      },
    });

    // Remove password from response
    const { contrasenia, ...userWithoutPassword } = user;
    return userWithoutPassword;
  }
}
```

### DTOs (Data Transfer Objects)

```typescript
// Use class-validator decorators
export class CreateUserDto {
  @IsString()
  @MinLength(2)
  @MaxLength(50)
  nombre: string;

  @IsString()
  @MinLength(2)
  @MaxLength(50)
  apellido: string;

  @IsEmail()
  mail: string;

  @IsString()
  @MinLength(6)
  @Matches(/^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)/, {
    message: 'Password must contain uppercase, lowercase, and number',
  })
  contrasenia: string;

  @IsOptional()
  @IsEnum(['usuario', 'admin'])
  rol?: string;
}

// Separate DTOs for different operations
export class UpdateUserDto {
  @IsOptional()
  @IsString()
  nombre?: string;

  @IsOptional()
  @IsString()
  apellido?: string;
}
```

### Guards

```typescript
// Keep guards focused and reusable
@Injectable()
export class RolesGuard implements CanActivate {
  constructor(private reflector: Reflector) {}

  canActivate(context: ExecutionContext): boolean {
    const requiredRoles = this.reflector.get<string[]>(
      'roles',
      context.getHandler(),
    );

    if (!requiredRoles) {
      return true;
    }

    const request = context.switchToHttp().getRequest();
    const user = request.user;

    return requiredRoles.includes(user?.rol);
  }
}
```

### Hooks

```jsx
// Custom hooks for reusable logic
import { useState, useEffect } from 'react';

/**
 * Custom hook for fetching data
 * @param {string} url API endpoint
 * @returns {Object} { data, loading, error }
 */
function useFetch(url) {
  const [data, setData] = useState(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    const fetchData = async () => {
      try {
        setLoading(true);
        const response = await fetch(url);
        const json = await response.json();
        setData(json);
      } catch (err) {
        setError(err);
      } finally {
        setLoading(false);
      }
    };

    fetchData();
  }, [url]);

  return { data, loading, error };
}

// Usage
function MyComponent() {
  const { data, loading, error } = useFetch('/api/users');

  if (loading) return <div>Loading...</div>;
  if (error) return <div>Error: {error.message}</div>;

  return <div>{/* Render data */}</div>;
}
```

## Database (Prisma)

### Schema Conventions

```prisma
// Use PascalCase for model names
model Usuario {
  id          Int       @id @default(autoincrement())
  nombre      String
  apellido    String
  mail        String    @unique
  contrasenia String
  rol         String    @default("usuario")
  
  // Relations - use descriptive names
  turnos      Turno[]
  resenias    Resenia[] @relation("ReseniasPorUsuario")
  
  // Timestamps
  createdAt   DateTime  @default(now())
  updatedAt   DateTime  @updatedAt

  // Indexes
  @@index([mail])
}

// Use descriptive field names
model Establecimiento {
  // Avoid abbreviations
  id                Int       @id @default(autoincrement())
  nombre            String    // Not 'nom' or 'name_est'
  direccion         String?   // Not 'addr' or 'dir'
  telefono          String?   // Not 'tel' or 'phone_num'
  
  // Use appropriate types
  lat               Float     // Geographic coordinates
  lng               Float
  horarios          String?   @db.Text  // Long text
  metadata          Json?     // JSON data
  
  createdAt         DateTime  @default(now())
  updatedAt         DateTime  @updatedAt
}
```

### Queries

```typescript
// Use descriptive query names
// Good
async function findUserByEmail(email: string) {
  return await prisma.usuario.findUnique({
    where: { mail: email },
  });
}

// Include related data explicitly
async function getUserWithAppointments(userId: number) {
  return await prisma.usuario.findUnique({
    where: { id: userId },
    include: {
      turnos: {
        include: {
          establecimiento: true,
          especialidad: true,
        },
        orderBy: { fecha: 'desc' },
      },
    },
  });
}

// Use transactions for multiple operations
async function createAppointmentWithNotification(data: CreateAppointmentDto) {
  return await prisma.$transaction(async (tx) => {
    const appointment = await tx.turno.create({ data });
    await tx.notification.create({
      data: {
        userId: data.usuarioId,
        message: 'Appointment created',
      },
    });
    return appointment;
  });
}
```
