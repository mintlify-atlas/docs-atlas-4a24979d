---
title: 'List User Reviews'
api: 'GET /resenias/mis-resenias'
---

Retrieves all reviews created by the authenticated user. This endpoint requires JWT authentication.

<Note>
  This endpoint requires authentication. Include a valid JWT token in the Authorization header.
</Note>

## Authentication

```bash
Authorization: Bearer <your_jwt_token>
```

The user ID is automatically extracted from the JWT token.

## Response

<ResponseField name="reviews" type="array">
  Array of review objects created by the authenticated user
  
  <ResponseField name="id" type="number">
    The unique identifier for the review
  </ResponseField>
  
  <ResponseField name="usuarioId" type="number">
    The ID of the user who created the review
  </ResponseField>
  
  <ResponseField name="turnoId" type="number">
    The ID of the appointment
  </ResponseField>
  
  <ResponseField name="establecimientoId" type="number">
    The ID of the healthcare establishment
  </ResponseField>
  
  <ResponseField name="puntuacion" type="number">
    Rating score (1-5)
  </ResponseField>
  
  <ResponseField name="comentario" type="string">
    The review comment
  </ResponseField>
  
  <ResponseField name="createdAt" type="string">
    Timestamp when the review was created
  </ResponseField>
  
  <ResponseField name="establecimiento" type="object">
    Establishment information
    
    <ResponseField name="id" type="number">
      Establishment ID
    </ResponseField>
    
    <ResponseField name="nombre" type="string">
      Establishment name
    </ResponseField>
  </ResponseField>
</ResponseField>
